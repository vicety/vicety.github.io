LeetCode 84、85 求直方图与二维矩阵中的最大矩形面积

vicety · 2018-12-12 · 次阅读

先说84题，要达到O(n)的复杂度，显然是单调增栈，遍历数组尝试插入，遇到违反单调性时对每个使得违反单调性的栈顶元素做退栈+更新答案操作直到这次插入不再违反单调性。

对于那些因为违反单调性将要被退栈的元素而言，由于违反单调性，以这个元素为最高高度的矩形的右边界已经被确定为i-1（令i为当前遍历的下标），左边界由于单调栈的特性也确定为这个栈中此元素下方存储的下标+1，此时以此位置为最高高度的矩形已经确定，因此可以计算答案+更新答案+退栈。

最后，为了确保栈中元素最后都能退栈并更新答案，数组最后要加入一个0

class Solution {
public:
    int largestRectangleArea(vector<int>& heights) {
        int res = 0;
        stack<int> st;
        heights.push_back(0);
        for (int i = 0; i < heights.size(); ++i) {
            while (!st.empty() && heights[st.top()] > heights[i]) {
                int cur = st.top(); st.pop();
                res = max(res, heights[cur] * (st.empty() ? i : (i - st.top() - 1)));
                cout << i <<  res << endl;
            }
            st.push(i);
        }
        return res;
    }
};

然后说85题，一种做法是遍历矩阵，每行统计一个height数组来记录这一行这一列最高的高度，统计完一行后，对这一行解84题中的问题，复杂度O(mn)

class Solution {
    private int solveLine(int[] heights) {
        int ans = 0;
        List<Integer> list = IntStream.of(heights).boxed().collect(Collectors.toList());
        list.add(0);
        Stack<Integer> st = new Stack<>();
        for (int i = 0; i < list.size(); i++) {
            while (!st.empty() && list.get(st.peek()) > list.get(i)) {
                int height = list.get(st.pop());
                ans = Math.max(ans, height * (st.empty() ? i : (i - st.peek() - 1)));
            }
            st.add(i);
        }
        return ans;
    }

    public int maximalRectangle(char[][] matrix) {
        if (matrix == null || matrix.length == 0 || matrix[0].length == 0) return 0;
        int row = matrix.length;
        int col = matrix[0].length;
        int[] height = new int[col];
        int ans = 0;
        for (int i = 0; i < row; i++) {
            for (int j = 0; j < col; j++) {
                if(matrix[i][j] == '1') height[j] = height[j] + 1;
                else height[j] = 0;
            }
            ans = Math.max(ans, solveLine(height));
        }
        return ans;
    }
}

另一种做法网上的题解我认为说得不够清晰，这也是为什么我要写关于这两道题的题解。先贴代码

class Solution {
public:
  int maximalRectangle(vector<vector<char>>& matrix) {
    int rows = matrix.size();
    if (rows == 0) return 0;
    int cols = matrix[0].size();
    if (cols == 0) return 0;
    vector<int> left(cols, 0);
    vector<int> right(cols, cols);
    vector<int> height(cols, 0);
    int area = 0;
    for (int i = 0; i < rows; ++i) {
      int current_left = 0;
      int current_right = cols;
      for (int j = 0; j < cols; ++j) {
        if (matrix[i][j] == '1') {
          ++height[j];
          left[j] = max(left[j], current_left);
        } else {
          height[j] = 0;
          left[j] = 0;
          current_left = j + 1;
        }
      }

      for (int j = cols - 1; j >=0; --j) {
        if (matrix[i][j] == '1') {
          right[j] = min(right[j], current_right);
        } else {
          right[j] = cols;
          current_right = j;
        }
      }
      for (int j = 0; j < cols; ++j)
        area = max(area, (right[j] - left[j]) * height[j]);
    }
    return area;
  }
};

left和right其实是对称关系，height维护的是当前位置向上数最多有几个1，left维护的是这个高度最左能到什么位置，right维护的是最右能到什么位置。每行的最后使用area = max(area, (right[j] - left[j]) * height[j])更新答案。另外，这里的left与right应该是二维dp数组优化后的结果。

分析left，current_left维护的是当前行、当前位置连续的1最左能到什么index，left[j]取决于current_left与上一行的left[j]，上一行的left[j]已经确保了维持当前位置高度能到达的最左位置，因此二者取更小的范围即可。另外，当当前位置为0时，由于height为0，这里的left right填什么都无法影响这一行的答案，但是需要考虑下一行，因此填一个最不限制范围的值防止影响下一行。

这种方法由于height需要一行行地加起来，不能用于高度值可能很大的84题（也许离散化可以搞一下？）。

最后放一个空间展开为二维的解法便于理解

public int maximalRectangle(char[][] matrix) {
    if (matrix == null || matrix.length == 0 || matrix[0].length == 0) return 0;
    int m = matrix.length;
    int n = matrix[0].length;
    int[][] height = new int[m][n];
    int[][] left = new int[m][n];
    int[][] right = new int[m][n];
    for (int i = 0; i < m; i++) {
        int currentLeft = 0;
        int currentRight = n;
        for (int j = 0; j < n; j++) {
            if (matrix[i][j] == '1') {
                if (i == 0) {
                    left[i][j] = currentLeft;
                    height[i][j] = 1;
                } else {
                    left[i][j] = Math.max(left[i - 1][j], currentLeft);
                    height[i][j] = height[i - 1][j] + 1;
                }
            } else {
                // height default zero
                currentLeft = j + 1;
                left[i][j] = 0;
            }
        }
        for (int j = n - 1; j >= 0; j--) {
            if (matrix[i][j] == '1') {
                if (i == 0) {
                    right[i][j] = currentRight;
                    height[i][j] = 1;
                } else {
                    right[i][j] = Math.min(right[i - 1][j], currentRight);
                    height[i][j] = height[i - 1][j] + 1;
                }
            } else {
                currentRight = j;
                right[i][j] = n;
            }
        }
    }
    int ans = 0;
    for (int i = 0; i < m; i++) {
        for (int j = 0; j < n; j++) {
            ans = Math.max(ans, height[i][j] * (right[i][j] - left[i][j]));
        }
    }
    return ans;
}
